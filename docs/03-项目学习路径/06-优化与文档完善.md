### 1.2 缓存优化

```python
# agent/cache.py
import redis
import json
from functools import wraps

redis_client = redis.Redis(host="cache", port=6379, decode_responses=True)

def cache(key_prefix: str, ttl: int = 3600):
    """缓存装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            cache_key = f"{key_prefix}:{str(args)}"
            
            # 尝试从缓存获取
            cached = redis_client.get(cache_key)
            if cached:
                return json.loads(cached)
            
            # 执行函数
            result = await func(*args, **kwargs)
            
            # 存入缓存
            redis_client.setex(
                cache_key,
                ttl,
                json.dumps(result)
            )
            
            return result
        return wrapper
    return decorator

# 使用示例
@cache("llm_response", ttl=1800)
async def get_llm_response(prompt: str) -> str:
    return await llm_manager.chat(prompt)
```

## 2. 错误处理

### 2.1 全局异常处理

```python
# api/exceptions.py
from fastapi import HTTPException, Request
from fastapi.responses import JSONResponse

class AgentException(Exception):
    """Agent 相关异常"""
    def __init__(self, message: str, code: str = "AGENT_ERROR"):
        self.message = message
        self.code = code

class ToolExecutionError(AgentException):
    """工具执行异常"""
    def __init__(self, tool_name: str, message: str):
        super().__init__(f"工具 {tool_name} 执行失败：{message}", "TOOL_ERROR")

class LLMRateLimitError(AgentException):
    """LLM 限流异常"""
    def __init__(self, retry_after: int = 60):
        super().__init__(f"LLM 请求限流，请在 {retry_after} 秒后重试", "RATE_LIMIT")
        self.retry_after = retry_after

# 注册全局异常处理器
async def agent_exception_handler(request: Request, exc: AgentException):
    return JSONResponse(
        status_code=400,
        content={
            "error": {
                "code": exc.code,
                "message": exc.message,
            }
        },
        headers={"Retry-After": str(getattr(exc, "retry_after", 60))}
    )

app.add_exception_handler(AgentException, agent_exception_handler)
```

### 2.2 重试机制

```python
# agent/retry.py
import asyncio
from functools import wraps

def retry(max_attempts: int = 3, delay: float = 1.0, backoff: float = 2.0):
    """指数退避重试装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            current_delay = delay
            last_exception = None
            
            for attempt in range(max_attempts):
                try:
                    return await func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    if attempt < max_attempts - 1:
                        await asyncio.sleep(current_delay)
                        current_delay *= backoff
                    else:
                        raise last_exception
        
        return wrapper
    return decorator

# 使用示例
@retry(max_attempts=3, delay=2.0, backoff=2.0)
async def call_llm_with_retry(prompt: str) -> str:
    return await llm_manager.chat(prompt)
```

## 3. 日志系统

### 3.1 结构化日志

```python
# agent/logger.py
import logging
import json
from datetime import datetime
from contextvars import ContextVar

# 请求追踪 ID
trace_id_var: ContextVar[str] = ContextVar("trace_id", default="")

class StructuredLogger:
    def __init__(self, name: str):
        self.logger = logging.getLogger(name)
    
    def log(self, level: str, event: str, **kwargs):
        log_data = {
            "timestamp": datetime.now().isoformat(),
            "trace_id": trace_id_var.get(),
            "event": event,
            **kwargs,
        }
        getattr(self.logger, level)(json.dumps(log_data))
    
    def info(self, event: str, **kwargs):
        self.log("info", event, **kwargs)
    
    def error(self, event: str, **kwargs):
        self.log("error", event, **kwargs)
    
    def warning(self, event: str, **kwargs):
        self.log("warning", event, **kwargs)

logger = StructuredLogger("ai_agent")

# 使用
logger.info("agent_start", agent_id="writer_001", task_id="task_123")
logger.error("agent_error", agent_id="writer_001", error="timeout")
```

## 6. 常见问题 FAQ

```markdown
## 常见问题

### Q: Agent 任务超时怎么办？
A: 检查 timeoutSeconds 配置，可以适当增大。如果频繁超时，说明任务太复杂，考虑拆分任务。

### Q: LLM API 限流怎么处理？
A: 系统内置了指数退避重试机制。可以在 .env 中配置多个 API Key 实现负载均衡。

### Q: 如何新增自定义工具？
A: 继承 BaseTool 类，实现 execute 方法，然后在 ToolRegistry 中注册即可。

### Q: 多 Agent 之间如何通信？
A: 通过 AgentRegistry 的 send_message 方法，支持点对点和广播模式。

### Q: 记忆系统如何持久化？
A: 短期记忆用 Redis，长期记忆用 PostgreSQL。向量检索使用 pg_vector 扩展。
```
